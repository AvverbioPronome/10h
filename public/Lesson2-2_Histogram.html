def make_da_plot(bins=None):
    fig = plt.figure(1,figsize=(15,10))
    labels = ["Uniform", "Normal", "Tri", "Exp", r"$\chi^2$", "Pow"]
    colors = "red blue orange black green purple".split(' ')

    Naxis=321

    for i in range(len(x)):
        ax=fig.add_subplot(Naxis)
        ax.hist(x[i], color=colors[i], bins=bins)
        ax.set_title(labels[i])
        Naxis += 1
    
    fig.suptitle("A collection of random data from several distributions.", fontsize=16)
    #fig.show() # avoid warnings about the matplotlib backend 
    
make_da_plot()
